~/.vimrc 
syntax enable
syntax on
colorscheme delek
filetype plugin on
set nocompatible

set number
set autoindent
set smartindent
set tabstop=4
set shiftwidth=4
set incsearch
set mouse=a

map <F4> :call DebugWithoutMess()<CR>
map <F5> :wa<CR>:make debug<CR>
map <F6> :wa<CR>
map <F9> :wa<CR>:!./configure<CR>
map <F12> gg=G

nmap <C-S-SPACE> VimwikiToggleListItem

let s:debugging = 0
function! DebugWithoutMess()
	if s:debugging == 0
		let s:debugging = 1
		wa
		new
		resize 1
		make debug
	else
		let s:debugging = 0
		q
	endif
endfunction

source ~/config/completion.vim

source $VIMRUNTIME/menu.vim
set wildmenu
set cpo-=<
set wcm=<C-Z>
map <F4> :emenu <C-Z>
